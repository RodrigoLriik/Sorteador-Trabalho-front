    <script>
        function sortear() {
            const maxNumber = document.getElementById("maxNumber").value;
            const resultadoDiv = document.getElementById("resultado");

            // Verifica se o número máximo é válido
            if (!maxNumber || Number(maxNumber) < 1) {
                alert("Por favor, insira um número máximo válido! Deve ser maior que 0.");
                return;
            }

            const resultados = [];
            for (let i = 0; i < 5; i++) {
                const sorteio = Math.floor(Math.random() * Number(maxNumber)) + 1;
                resultados.push(sorteio);
            }

            // Exibe os resultados em uma string formatada
            resultadoDiv.textContent = `Números sorteados: ${resultados.join(', ')}`;
            resultadoDiv.style.opacity = 1;

            // Adiciona animação de fade-in no resultado
            setTimeout(() => {
                resultadoDiv.style.opacity = 0;
            }, 3000);
        }
    </script>
